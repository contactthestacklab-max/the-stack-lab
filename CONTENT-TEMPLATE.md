## Code à copier dans le fichier .astro

```astro
---
import ProductLayout from '../layouts/ProductLayout.astro';

const productConfig = {
  // SEO
  title: "[VOTRE TITRE]",
  description: "[VOTRE DESCRIPTION]",

  // Hero
  headline: "[VOTRE HEADLINE]",
  heroImage: "/images/[votre-image].webp",

  // Produit
  productName: "[NOM DU PRODUIT]",
  affiliateLink: "[VOTRE LIEN SYSTEME.IO]",

  // Problem
  problemTitle: "[TITRE PROBLEM]",
  problemText: `
    <p>[PARAGRAPHE 1]</p>
    <p>[PARAGRAPHE 2]</p>
    <p>[PARAGRAPHE 3]</p>
  `,

  // Agitation
  agitationTitle: "[TITRE AGITATION]",
  painPoints: [
    "[POINT 1]",
    "[POINT 2]",
    "[POINT 3]",
    "[POINT 4]",
  ],

  // Solution
  solutionTitle: "[TITRE SOLUTION]",
  solutionText: "[TEXTE INTRO SOLUTION]",
  benefits: [
    { title: "[AVANTAGE 1]", description: "[DESCRIPTION 1]" },
    { title: "[AVANTAGE 2]", description: "[DESCRIPTION 2]" },
    { title: "[AVANTAGE 3]", description: "[DESCRIPTION 3]" },
  ],

  // Témoignages
  testimonials: [
    { name: "[PRENOM 1]", rating: 5, text: "[TEXTE 1]" },
    { name: "[PRENOM 2]", rating: 5, text: "[TEXTE 2]" },
    { name: "[PRENOM 3]", rating: 4, text: "[TEXTE 3]" },
    { name: "[PRENOM 4]", rating: 5, text: "[TEXTE 4]" },
  ],

  // Comparatif
  comparisonItems: [
    { feature: "[CRITÈRE 1]", product: true, competitors: false },
    { feature: "[CRITÈRE 2]", product: true, competitors: false },
    { feature: "[CRITÈRE 3]", product: "[VALEUR]", competitors: "[VALEUR]" },
    { feature: "[CRITÈRE 4]", product: true, competitors: false },
    { feature: "[CRITÈRE 5]", product: true, competitors: false },
  ],

  // CTA
  ctaText: "[TEXTE BOUTON]",
  finalCtaTitle: "[TITRE CTA FINAL]",
  finalCtaText: "[TEXTE CTA FINAL]",
};
---

<ProductLayout {...productConfig} />
```
